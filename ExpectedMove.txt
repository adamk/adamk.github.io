declare upper;

input showEM = yes;
input sessionMinutes = 390;        # 09:30 -> 16:00 = 390
input useRTHOpenAnchor = yes;      # if no, uses prior close
input showLabels = yes;

# --- New day ---
def newDay = GetYYYYMMDD() <> GetYYYYMMDD()[1];

# --- RTH session flags (constants only) ---
def isRTH = SecondsFromTime(0930) >= 0 and SecondsTillTime(1600) > 0;
def rthStart = isRTH and !isRTH[1];
  
# --- Anchor price (latched) ---
def rthOpenLatched =
    CompoundValue(1,
        if newDay then Double.NaN
        else if rthStart then open
        else rthOpenLatched[1],
        Double.NaN
    );

def priorClose = close(period = AggregationPeriod.DAY)[1];
def anchor = if useRTHOpenAnchor then rthOpenLatched else priorClose;

# --- Real IV from Thinkorswim (decimal, e.g., 0.18 = 18%) ---
def iv = ImpVolatility();

# --- Time fraction in years ---
def T = sessionMinutes / 525600.0;

# --- Expected Move (1σ) ---
def em = if !IsNaN(anchor) and !IsNaN(iv) then anchor * iv * Sqrt(T) else Double.NaN;
def emUpper = anchor + em;
def emLower = anchor - em;

plot EM_Upper = if showEM then emUpper else Double.NaN;
plot EM_Lower = if showEM then emLower else Double.NaN;

EM_Upper.SetLineWeight(2);
EM_Lower.SetLineWeight(2);
EM_Upper.SetStyle(Curve.FIRM);
EM_Lower.SetStyle(Curve.FIRM);

AddLabel(showLabels and showEM and !IsNaN(em),
    "EM ±1σ: " + AsDollars(em) + " | IV: " + AsPercent(iv) + " | Anchor: " + AsDollars(anchor),
    Color.GRAY
);
EM_Upper.SetDefaultColor(Color.CYAN);
EM_Lower.SetDefaultColor(Color.CYAN);
EM_Upper.SetLineWeight(2);
EM_Lower.SetLineWeight(2);
